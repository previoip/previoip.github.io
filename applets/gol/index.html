 43class ac2DMatrix {
 44  itox(i) {return i % this.w};
 45  itoy(i) {return ftoi(i / this.w)};
 46  ctoi(x, y) {return x + (y * this.w)};
 47  get(x, y) {return this.buf[this.ctoi(x, y)]};
 48  set(x, y, v) {this.buf[this.ctoi(x, y)] = v;};
 49  map(cb) {for (let i=0; i<this.l; i++){this.buf[i] = cb(this.buf[i])}};
 50  zeros() {for (let i=0; i<this.l; i++) {this.buf[i] = 0}};
 51  sum() {let s=0; for (let i=0; i<this.l; i++) {s+=this.buf[i]}; return s};
 52  count(o) {let s=0; for (let i=0; i<this.l; i++) {if(this.buf[i]===o){s+=1}}; return s};
 53  load2DArray(arr) {
 54    for (let x=0; x<this.w; x++) {
 55      for (let y=0; y<this.h; y++) {
 56        this.set(x, y, arr[y][x]);
 57      }
 58    }
 59  };
 60  to2DArray() {
 61    let r = []
 62    for (let y=0; y<this.h; y++) {
 63      r.push([]);
 64      for (let x=0; x<this.w; x++) {
 65        r[y].push(this.get(x, y));
 66      }
 67    };
 68    return r;
 69  };
 70  queryKernel(mk, x, y, wrap) {
 71    const lx = ftoi(mk.w/2);
 72    const ly = ftoi(mk.h/2);
 73    for (let kx=0; kx<mk.w; kx++) {
 74      for (let ky=0; ky<mk.h; ky++) {
 75        let dx = x+kx-lx;
 76        let dy = y+ky-ly;
 77        if (wrap||false) {
 78          dx = iwrap(dx, this.w)
 79          dy = iwrap(dy, this.h)
 80        };
 81        mk.set(kx, ky, this.get(dx, dy) || null)
 82      }
 83    }
 84  };

 97class ParticleSim {
 98  constructor (w, h) {
 99    this.w = w; this.h = h, this.l=this.w*this.h;
100    this.zb = [new cUint8Matrix(this.w, this.h), new cUint8Matrix(this.w, this.h)];
101    this.z = this.zb[0];
102    this.k3 = new cUint8Matrix(3, 3);
103    this.rules = {};
104    this.colors = ["white"];
105    this.renderQueue = Array.from({length:this.colors.length+1}, ()=>[])
106  };
107  swpz() {
108    this.z = this.zb[1];
109    this.zb[1] = this.zb[0];
110    this.zb[0] = this.z;
111  };
112  update() {
113    let a = this.zb[0];
114    let b = this.zb[1];
115    let v = null;
116    b.zeros();
117    for (let n=0; n<this.renderQueue.length; n++) {
118      this.renderQueue[n] = [];
119    }
120    for (let i=0; i<a.l; i++) {
121      a.queryKernel(this.k3, a.itox(i), a.itoy(i), true);
122      v = this.k3.get(1,1);
123      v = this.rules[v](this.k3);
124      if (v>0) {
125        this.renderQueue[v].push(i);
126      }
127      b.set(b.itox(i), b.itoy(i), v);
128    };
129    this.swpz();
130  };
131  setCell(x, y, v) {
132    this.z.set(iwrap(x, this.w), iwrap(y, this.h), v);
133    this.renderQueue[v].push(this.z.ctoi(x, y))
134  };
135  render(ctx, w, h) {
136    const rx = (w-1) / this.w;
137    const ry = (h-1) / this.h;
138    const px = 2;
139    var i;
140    ctx.clearRect(0,0,w,h);
141    for (let n=0; n<this.renderQueue.length; n++) {
142      if (this.renderQueue[n].length > 0) {
143        ctx.beginPath();
144        ctx.fillStyle = this.colors[(n+1)%this.colors.length]
145        while (this.renderQueue[n].length > 0) {
146          i = this.renderQueue[n].pop();
147          ctx.rect(rx*this.z.itox(i)+px, ry*this.z.itoy(i)+px, rx-px, ry-px)
148        };
149        ctx.fill();
150      }
151    };
152  };
153  reset() {
154    this.zb[0].zeros();
155    this.zb[1].zeros();
156  };
157};
