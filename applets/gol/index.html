  9class acAbstract2DMatrix {
 10  itox(i) {return i % this.w};
 11  itoy(i) {return ftoi(i / this.w)};
 12  ctoi(x, y) {return x + (y * this.w)};
 13  get(x, y) {return this.buf[this.ctoi(x, y)]};
 14  set(x, y, v) {this.buf[this.ctoi(x, y)] = v;};
 15  map(cb) {for (let i=0; i<this.l; i++){this.buf[i] = cb(this.buf[i])}};
 16  zeros() {for (let i=0; i<this.l; i++) {this.buf[i] = 0}};
 17  sum() {let s=0; for (let i=0; i<this.l; i++) {s+=this.buf[i]}; return s};
 18  count(o) {let s=0; for (let i=0; i<this.l; i++) {if(this.buf[i]===o){s+=1}}; return s};
 19  load2DArray(arr) {
 20    for (let x=0; x<this.w; x++) {
 21      for (let y=0; y<this.h; y++) {
 22        this.set(x, y, arr[y][x]);
 23      }
 24    }
 25  };
 26  to2DArray() {
 27    let r = []
 28    for (let y=0; y<this.h; y++) {
 29      r.push([]);
 30      for (let x=0; x<this.w; x++) {
 31        r[y].push(this.get(x, y));
 32      }
 33    };
 34    return r;
 35  };
 36  queryKernel(mk, x, y, wrap) {
 37    const lx = ftoi(mk.w/2);
 38    const ly = ftoi(mk.h/2);
 39    for (let kx=0; kx<mk.w; kx++) {
 40      for (let ky=0; ky<mk.h; ky++) {
 41        let dx = x+kx-lx;
 42        let dy = y+ky-ly;
 43        if (wrap||false) {
 44          dx = dx<0?(this.w+(dx%this.w)):dx%this.w;
 45          dy = dy<0?(this.h+(dy%this.h)):dy%this.h;
 46        };
 47        mk.set(kx, ky, this.get(dx, dy) || null)
 48      }
 49    }
 50  };

105class ParticleSim {
106  constructor (w, h) {
107    this.w = w; this.h = h, this.l=this.w*this.h;
108    this.zb = [new cUint8Matrix(this.w, this.h), new cUint8Matrix(this.w, this.h)];
109    this.z = this.zb[0];
110    this.k = new cUint8Matrix(3, 3);
111    this.rules = {};
112    this.colors = ["white"];
113    this.renderQueue = Array.from({length:this.colors.length+1}, ()=>[])
114  };
115  swpz() {
116    this.z = this.zb[1];
117    this.zb[1] = this.zb[0];
118    this.zb[0] = this.z;
119  };
120  update() {
121    let a = this.zb[0];
122    let b = this.zb[1];
123    let v = null;
124    b.zeros();
125    for (let n=0; n<this.renderQueue.length; n++) {
126      this.renderQueue[n] = [];
127    }
128    for (let i=0; i<a.l; i++) {
129      a.queryKernel(this.k, a.itox(i), a.itoy(i), true);
130      v = this.k.get(1,1);
131      v = this.rules[v](this.k);
132      if (v>0) {
133        this.renderQueue[v].push(i);
134      }
135      b.set(b.itox(i), b.itoy(i), v);
136    };
137    this.swpz();
138  };
139  setCell(x, y, v) {
140    this.z.set(x, y, v);
141    this.renderQueue[v].push(this.z.ctoi(x, y))
142  };
143  render(ctx, w, h) {
144    const rx = w / this.w;
145    const ry = h / this.h;
146    const px = 2;
147    var i;
148    ctx.clearRect(0,0,w,h);
149    for (let n=0; n<this.renderQueue.length; n++) {
150      if (this.renderQueue[n].length > 0) {
151        ctx.beginPath();
152        ctx.fillStyle = this.colors[(n+1)%this.colors.length]
153        while (this.renderQueue[n].length > 0) {
154          i = this.renderQueue[n].pop();
155          ctx.rect(rx*this.z.itox(i)+px, ry*this.z.itoy(i)+px, rx-px, ry-px)
156        };
157        ctx.fill();
158      }
159    };
160  };
161  reset() {
162    this.zb[0].zeros();
163    this.zb[1].zeros();
164  };
165};
