 22function f_p(samples, p) {
 23  const n = p.length;
 24  let sums = 0;
 25  return samples.map(
 26    (s)=>{
 27        sums = 0;
 28        for (let m=0; m<n; m++) {
 29          sums += s**(n-m-1) * p[m];
 30        }
 31        sums = sums > 1 ? 1 : sums;
 32        sums = sums < 0 ? 0 : sums;
 33        return sums
 34    }
 35  )
 36};

144function linterp(a, b, t) {
145  return a + ((b - a) * t)
146};

149function arrLinterp(arr, t) {
150  l = arr.length || arr.byteLength;
151  f = (l-1)*t;
152  n = ftoi(f);
153  d = f % 1;
154  a = arr[n];
155  b = n+1 > l-1 ? a : arr[n+1];
156  return linterp(a, b, d);
157};

162function cTimer () {
163  this.f = 0;
164  this.fps = 0;
165  this.t = 0;
166  this.pt = 0;
167  this.dt = 0;
168  this.update = () => {
169    this.f += 1;
170    this.t = Date.now();
171    this.dt = this.t - this.pt;
172    this.pt = this.t;
173    this.fps = this.dt > 0 ? ftoi(1 / this.dt * 1000) : NaN;
174  }
175  this.progress = (n_loop) => this.f % n_loop / Math.max(n_loop-1, 1)
176  this.progress_t = (t_loop) => this.t % t_loop / Math.max(t_loop-1, 1)
177};

281  }
