 45class Interp {
 46  static lk2(wa, ha, wb, hb, callback) {
 47    let rx, ry, ax, ay, bx, by, bxl, bxh, byl, byh, wx, wy;
 48    rx = (wa-1) / (wDst-1);
 49    ry = (wa-1) / (hDst-1);
 50    for (ax=0; ax<wDst; ax++) {
 51      for (ay=0; ay<hDst; ay++) {
 52        bxl = Math.floor(rx * ax);
 53        bxr = Math.ceil (rx * ax);
 54        byl = Math.floor(ry * ay);
 55        byr = Math.ceil (ry * ay);
 56        wx = rx*ax-bxl;
 57        wy = ry*ay-byl;
 58        callback(
 59          x, y,
 60          bxl+wSrc*byl, bxl+wSrc*byr,
 61          bxr+wSrc*byl, bxr+wSrc*byr,
 62          wx, wy
 63        );
 64      };
 65    };
 66  };
 67  static nearest(a, b, t) {
 68    return t >= .5 ? b : a;
 69  };
 70  static binearest(a1, b1, a2, b2, t1, t2) {
 71    return Interp.nearest(
 72      Interp.nearest(a1, b1, t1),
 73      Interp.nearest(a2, b2, t1),
 74      t2
 75    )
 76  };
 77  static binearest_a(arrDst, wDst, hDst, arrSrc, wSrc, hSrc) {
 78    Interp.lk2(wDst, hDst, wSrcm hSrc, (x, y xlyl, xlyh, xhyl, xhyh, wx, wy)=>{
 79      arrDst[x*wDst+y] = Interp.binearest(
 80        arrSrc[xlyl], arrSrc[xhyl],
 81        arrSrc[xlyh], arrSrc[xhyh],
 82        wx, wy
 83      );
 84    });
 85  };
 86  static binearest_m(mDst, mSrc) {
 87    Interp.binearest_a(mDst.buf, mDst.w, mDst.h,
 88                      mSrc.buf, mSrc.w, mSrc.h);
 89  };
 90  static linear(a, b, t) {
 91    return a + (b - a) * t
 92  };
 93  static bilinear(a1, b1, a2, b2, t1, t2) {
 94    return Interp.linear(
 95      Interp.linear(a1, b1, t1),
 96      Interp.linear(a2, b2, t1),
 97      t2
 98    )
 99  };
100  static bilinear_a(arrDst, wDst, hDst, arrSrc, wSrc, hSrc) {
101    let ax, ay, rx, ry, wx, wy, bxl, bxr, byl, byr;
102    rx = (wSrc-1) / (wDst-1);
103    ry = (hSrc-1) / (hDst-1);
104    for (ax=0; ax<wDst; ax++) {
105      for (ay=0; ay<hDst; ay++) {
106        bxl = Math.floor(rx * ax);
107        bxr = Math.ceil (rx * ax);
108        byl = Math.floor(ry * ay);
109        byr = Math.ceil (ry * ay);
110        wx = rx*ax-bxl;
111        wy = ry*ay-byl;
112        arrDst[ax+ay*wDst] = Interp.bilinear(
113          arrSrc[bxl+wSrc*byl], arrSrc[bxr+wSrc*byl],
114          arrSrc[bxl+wSrc*byr], arrSrc[bxr+wSrc*byr],
115          wx, wy
116        );
117      }
118    };
119  };

127class cFloat32Matrix {
128  constructor (w, h) {
129    this.w = w; this.h = h; this.l = w*h;
130    this.buf = new Float32Array(this.l);
131  }
132  itox(i) {return i % this.w};
133  itoy(i) {return ftoi(i / this.w)};
134  ctoi(x, y) {return x + (y * this.w)};
135  get(x, y) {return this.buf[this.ctoi(x, y)]};
136  set(x, y, v) {this.buf[this.ctoi(x, y)] = v;};
137  add(v) {for (let i=0; i<this.l; i++){this.buf[i] = this.buf[i] + v}};
138  mul(v) {for (let i=0; i<this.l; i++){this.buf[i] = this.buf[i] * v}};
139  map(callback) {for (let i=0; i<this.l; i++){this.buf[i] = callback(this.buf[i])}};
140  zeros() {for (let i=0; i<this.buf.length; i++) {this.buf[i] = 0}};
141  load2DArray(arr) {
142    for (let x=0; x<this.w; x++) {
143      for (let y=0; y<this.h; y++) {
144        this.set(x, y, arr[y][x]);
145      }
146    }
147  };
148  loadImageData(data) {
149    for (let i=0; i<data.length; i+=4) {
150      this.buf[ftoi(i/4)] = (data[i+0] + data[i+1] + data[i+2]) / 3 / 255;
151    }
152  };
153  unloadImageData(data) {
154    for (let i=0; i<data.length; i+=4) {
155      let p = this.buf[ftoi(i/4)];
156      p = p > 1 ? 1 : p;
157      p = p < 0 ? 0 : p;
158      p = ftoi(p * 255);
159      data[i+0] = p;
160      data[i+1] = p;
161      data[i+2] = p;
162    }
163  };
164  upscaleInset(m) {
165    let scale = ftoi(m.w / this.w);
166    for (let x=0; x<this.w; x++) {
167      for (let y=0; y<this.h; y++) {
168        for (let i=0; i<scale; i++) {
169          for (let j=0; j<scale; j++) {
170            m.set((x*scale)+i, (y*scale)+j, this.get(x, y));
171          }
172        }
173      }
174    }
175  };
176}

273function conv2d (mSrc, mDst, mKernel, padding, stride, callback) {
274  let ki, kx, ky, oi, ox, oy, ix, iy;
275  for (oi=0; oi<mDst.l; oi++){
276    ox = oi%mDst.w;
277    oy = Math.floor(oi/mDst.w);
278    for (ki=0; ki<mKernel.l; ki++){
279      kx = ki%mKernel.w;
280      ky = Math.floor(ki/mKernel.w);
281      ix=(ox*stride)+kx-padding; iy=(oy*stride)+ky-padding;
282      callback(mSrc, mDst, mKernel, ix, iy, kx, ky, ox, oy);
283    }
284  }
285}
